Quick setup for new users

# Clone
git clone <your-repo-url>
cd <repo-name>

# Backend setup
cd news-backend
npm install
npm run dev

# Frontend setup (in new terminal)
cd news-aggregator
npm install
npm start

# Fetch news (in new terminal)
cd news-backend
node updateNews.js
